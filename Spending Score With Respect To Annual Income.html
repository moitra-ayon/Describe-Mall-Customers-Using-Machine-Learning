removeNA = function(){
  
  mall[4] <<- ifelse(is.na(mall$Annual.Income..k..),
                  mean(mall$Annual.Income..k..,na.rm = TRUE),
                  mall$Annual.Income..k..)
  
  mall[5] <<- ifelse(is.na(mall$Spending.Score..1.100.),
                  mean(mall$Spending.Score..1.100.,na.rm = TRUE),
                  mall$Spending.Score..1.100.)
  mall  
}

prediction1 = function(){
  set.seed(999)
  res <<- kmeans(train.data[,4:5], centers = 5, iter.max = 12, nstart = 2)
  res$size 
  
  res$centers
  
  res$cluster
  
  plot(train.data[,4:5],col=res$cluster)  
    
  print(paste("Accuracy ",res$betweenss/res$totss*100))    
}
